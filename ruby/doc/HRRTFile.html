def initialize(match)
  if match.nil?
    raise
  else
    @subject_name_last  = match[:last].upcase
    @subject_name_first = match[:first].upcase
    @subject_history    = match[:hist].upcase
    @scan_date  = match.names.include?('date') ? match[:date] : make_date(match)
    @scan_time  = match.names.include?('time') ? match[:time] : make_time(match)
    @scan_type  = match[:type].upcase
    @scan_extn  = match[:extn].downcase
    @datetime   = @scan_date + '_' + @scan_time
  end
end

def self.matches_name(infile)

def make_date(match)
  if (match.names & NEEDED_NAMES_DATE).sort.eql?(NEEDED_NAMES_DATE.sort)
    date = sprintf(HRRT_DATE_FMT, yr: match[:yr].to_i % 100, mo: match[:mo], dy: match[:dy])
  else
    raise
  end
  date
end

def print_summary(short = false)
  if short
    printf("%-8s%-8s%-10s%-12s%-12s%-4s%-8s%10d%15s\n", @scan_date, @scan_time, @subject_history, @subject_name_first, @subject_name_last, @scan_type, @scan_extn, @file_size, @file_modified.strfscan_time("%y%m%d_%H%M%S"))
  else
    puts("scan_time: #{time}")
    puts("scan_date: #{scan_date}")
    puts("subject_history: #{subject_history}")
    puts("subject_name_first: #{subject_name_first}")
    puts("subject_name_last: #{subject_name_last}")
  end
end

def read_physical(infile)
  stat = File.stat(infile)
  @file_name = File.basename(infile)
  @file_path = File.dirname(infile)
  @file_size = stat.size
  @file_modified = stat.mtime
end
