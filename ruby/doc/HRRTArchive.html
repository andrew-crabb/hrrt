def archive_file(f)
  name_in_archive = name_in_archive(f)
  unless present?(name_in_archive)
    store_file(f)
  end
end

def archive_files(acs)
  acs.scans_by_datetime.each do |datetime, scan|
    scan.files.each { |f| archive_file(f) }
  end
end
