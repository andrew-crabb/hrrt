def initialize(scanfiles)
  mylogger.info("initialize")
  @files = {}
  scanfiles.each { |f| @files[f.scan_extn] = f }
  @has_all_files = has_all_files
  @subject = HRRTSubject.new(@files[HRRTFile::EXTN_L64_HDR])
end
