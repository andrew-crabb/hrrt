def initialize(indir)
  mylogger.info("initialize(#{indir})")
  @hrrt_files = {}
  @indir = indir
  Dir.chdir(indir)
  infiles = Dir.glob("**/*").select {|f| File.file? f}
  infiles.each do |infile|
    if (match = HRRTFile.matches_name(infile))
      @hrrt_files[infile] = HRRTFile.new(match)
      @hrrt_files[infile].read_physical(File.join(indir, infile))
    end
  end
end

def make_files_by_datetime
  unless defined? @files_by_datetime
    @files_by_datetime = {}
    @hrrt_files.each do |filename, hrrt_file|
      @files_by_datetime[hrrt_file.datetime] ||= Array.new
      @files_by_datetime[hrrt_file.datetime].push(hrrt_file)
    end
  end
end
