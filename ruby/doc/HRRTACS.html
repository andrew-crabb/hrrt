def read_dirs(indir)
  @all_files = []
  Dir.chdir(indir)
  all_dirs = Dir.glob('*').select { |f| File.directory? f }

  @all_files = Dir.glob("**/*").select { |f| File.file? f }
  all_dirs = Dir.glob('*').select { |f| File.directory? f }
  all_dirs.each do |subdir|
    read_dir(subdir)
  end
end

def scan_dirs(indir)
  @acs_dirs = {}
  Dir.chdir(indir)
  all_dirs = Dir.glob('*').select { |f| File.directory? f }
  all_dirs.each do |subdir|
    @acs_dirs[subdir] = HRRTACSDir.new(File.join(indir, subdir))
    @acs_dirs[subdir].create_hrrt_files
  end
end
